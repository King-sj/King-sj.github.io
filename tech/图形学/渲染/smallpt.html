void renderProgressBar(int y, int h, int samps) {
  int barWidth = 70;
  double progress = (double)y / (h - 1);
  int pos = barWidth * progress;

  std::cout << "\r[";
  for (int i = 0; i < barWidth; ++i) {
    if (i < pos)
      std::cout << "=";
    else if (i == pos)
      std::cout << ">";
    else
      std::cout << " ";
  }
  std::cout << "] " << int(progress * 100.0) << " % (" << samps * 4 << " spp)";
  std::cout.flush();
}
